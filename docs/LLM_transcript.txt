from typing import Dict, List, Union

Number = Union[int, float]


def is_allocation_feasible(
    resources: Dict[str, Number],
    requests: List[Dict[str, Number]]
) -> bool:

    if not isinstance(resources, dict):
        raise ValueError("Resources must be a dictionary.")

    total_usage: Dict[str, Number] = {r: 0 for r in resources}

    for request in requests:
        if not isinstance(request, dict):
            raise ValueError("Each request must be a dictionary.")

        for resource_name, amount in request.items():
            if resource_name not in resources:
                return False

            if not isinstance(amount, (int, float)):
                raise ValueError("Resource amounts must be numeric.")

            if amount < 0:
                return False

            total_usage[resource_name] += amount

    for resource_name, used in total_usage.items():
        if used > resources[resource_name]:
            return False

    return True

LLM Response 3:
The function takes two input parameters.

resources: Type Dict[str, Number]. It is a dictionary mapping resource names (strings) to their total available capacity (non-negative integers or floats). Example: {'cpu': 10, 'mem': 32}.

requests: Type List[Dict[str, Number]]. It is a list of dictionaries where each dictionary represents a request mapping resource names to required amounts (non-negative integers or floats). Each referenced resource must exist in resources. Example: [{'cpu': 2, 'mem': 8}, {'cpu': 3, 'mem': 10}].
